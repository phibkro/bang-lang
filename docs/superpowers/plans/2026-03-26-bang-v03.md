# Bang v0.3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add match+patterns, type declarations (ADTs), mut+assignment, and import/export to the Bang compiler.

**Architecture:** Each feature follows the pipeline: AST node → lexer → parser → interpreter → checker → codegen → formatter → AstGen → tests. Features are ordered by dependency: type declarations first (constructors needed for match patterns), then match, then mut, then import/export.

**Tech Stack:** TypeScript, Effect (Schema.TaggedClass, Match.exhaustive, HashMap, Data.taggedEnum), @effect/vitest, @effect/printer

**Spec:** `docs/superpowers/specs/2026-03-26-bang-v03-design.md`

```typescript
const parseTypeDecl = (s: ParseState): P<Ast.TypeDecl> =>
  Effect.gen(function* () {
    const [startTok, s1] = yield* expect(s, "Keyword", "type");
    const [nameTok, s2] = yield* expect(s1, "TypeIdent");
    // Collect type params (lowercase idents before =)
    const go = (params: string[], st: ParseState): P<string[]> =>
      Effect.gen(function* () {
        const isEq = yield* check(st, "Operator", "=");
        if (isEq) return [params, st] as const;
        const [tok, st2] = yield* expect(st, "Ident");
        return yield* go([...params, tokenValue(tok)], st2);
      });
    const [typeParams, s3] = yield* go([], s2);
    const [, s4] = yield* expect(s3, "Operator", "=");
    // Parse constructors separated by |
    const [firstCtor, s5] = yield* parseConstructor(s4);
    const ctors = (ctorList: Ast.Constructor[], st: ParseState): P<Ast.Constructor[]> =>
      Effect.gen(function* () {
        const isPipe = yield* check(st, "Delimiter", "|");
        if (!isPipe) return [ctorList, st] as const;
        const [, st2] = yield* advance(st); // consume |
        const [ctor, st3] = yield* parseConstructor(st2);
        return yield* ctors([...ctorList, ctor], st3);
      });
    const [constructors, s6] = yield* ctors([firstCtor], s5);
    return [
      new Ast.TypeDecl({
        name: tokenValue(nameTok),
        typeParams,
        constructors,
        span: Span.merge(tokenSpan(startTok), constructors[constructors.length - 1].span),
      }),
      s6,
    ] as const;
  });
```

Implement `parseConstructor`:

```typescript
const parseConstructor = (s: ParseState): P<Ast.Constructor> =>
  Effect.gen(function* () {
    const [nameTok, s1] = yield* expect(s, "TypeIdent");
    const tag = tokenValue(nameTok);
    const nameSpan = tokenSpan(nameTok);
    // Check for named fields: { ... }
    const isBrace = yield* check(s1, "Delimiter", "{");
    if (isBrace) {
      const [, s2] = yield* advance(s1); // consume {
      const fields: Array<{ name: string; type: Ast.Type }> = [];
      const parseFields = (st: ParseState): P<typeof fields> =>
        Effect.gen(function* () {
          const isClose = yield* check(st, "Delimiter", "}");
          if (isClose) return [fields, st] as const;
          if (fields.length > 0) {
            const [, st2] = yield* expect(st, "Delimiter", ",");
            st = st2;
          }
          const [fieldName, st3] = yield* expect(st, "Ident");
          const [, st4] = yield* expect(st3, "Delimiter", ":");
          const [fieldType, st5] = yield* parseType(st4);
          fields.push({ name: tokenValue(fieldName), type: fieldType });
          return yield* parseFields(st5);
        });
      const [namedFields, s3] = yield* parseFields(s2);
      const [endTok, s4] = yield* expect(s3, "Delimiter", "}");
      return [
        new Ast.NamedConstructor({
          tag,
          fields: namedFields,
          span: Span.merge(nameSpan, tokenSpan(endTok)),
        }),
        s4,
      ] as const;
    }
    // Check for positional fields: type variables that were declared as type params
    // IMPORTANT: Only accept Ident tokens that match declared type params of the
    // current type declaration, or TypeIdent tokens. This prevents greedily consuming
    // the next statement's identifier as a positional field.
    // E.g., in "type Maybe a = Some a | None\nx = 42", we must not consume "x" as a field of None.
    // Pass typeParamNames to this function from the caller.
    const types: Ast.Type[] = [];
    const parsePositionalFields = (st: ParseState, typeParamNames: Set<string>): P<Ast.Type[]> =>
      Effect.gen(function* () {
        const atEnd = yield* isAtEnd(st);
        if (atEnd) return [types, st] as const;
        const next = yield* peek(st);
        // Stop at | or ; or } or EOF or any keyword or operator
        if (
          tokenTag(next) === "Delimiter" &&
          (tokenValue(next) === "|" || tokenValue(next) === ";" || tokenValue(next) === "}")
        )
          return [types, st] as const;
        if (tokenTag(next) === "EOF") return [types, st] as const;
        if (tokenTag(next) === "Keyword") return [types, st] as const;
        if (tokenTag(next) === "Operator") return [types, st] as const;
        // Accept TypeIdent (concrete type names like Int, String)
        if (tokenTag(next) === "TypeIdent") {
          const [typ, st2] = yield* parsePrimaryType(st);
          types.push(typ);
          return yield* parsePositionalFields(st2, typeParamNames);
        }
        // Accept Ident ONLY if it's a declared type parameter of this type decl
        if (tokenTag(next) === "Ident" && typeParamNames.has(tokenValue(next))) {
          const [typ, st2] = yield* parsePrimaryType(st);
          types.push(typ);
          return yield* parsePositionalFields(st2, typeParamNames);
        }
        // Any other token: stop (it's the start of the next statement)
        return [types, st] as const;
      });
    const typeParamSet = new Set(/* pass typeParams from the caller */);
    const [positionalFields, s2] = yield* parsePositionalFields(s1, typeParamSet);
    if (positionalFields.length === 0) {
      return [new Ast.NullaryConstructor({ tag, span: nameSpan }), s2] as const;
    }
    return [
      new Ast.PositionalConstructor({
        tag,
        fields: positionalFields,
        span: Span.merge(nameSpan, positionalFields[positionalFields.length - 1].span),
      }),
      s2,
    ] as const;
  });
```

```typescript
export const evalProgram = (program: Ast.Program): Effect.Effect<Value, EvalError> =>
  Effect.gen(function* () {
    let env: Env = HashMap.empty();
    let lastValue: Value = Unit();

    for (const stmt of program.statements) {
      env = yield* evalStmt(stmt, env);
      // Extract lastValue from the updated env
      if (stmt._tag === "Declaration") {
        lastValue = yield* evalExpr(stmt.value, env);
      } else if (stmt._tag === "ForceStatement") {
        lastValue = yield* evalExpr(stmt.expr, env);
      } else if (stmt._tag === "ExprStatement") {
        lastValue = yield* evalExpr(stmt.expr, env);
      } else {
        // TypeDecl, Declare, Import, Export, Mutation — don't change lastValue
        // (except Mutation could, but for v0.3 we keep it simple)
      }
    }

    return lastValue;
  });
```

```typescript
  Effect.gen(function* () {
    const [startTok, s1] = yield* expect(s, "Keyword", "match");
    const [scrutinee, s2] = yield* parseExprPrec(s1, PREC_APP + 1); // parse scrutinee without consuming arms
    const [, s3] = yield* expect(s2, "Delimiter", "{");

    const arms: Ast.Arm[] = [];

      Effect.gen(function* () {
        const isClose = yield* check(st, "Delimiter", "}");
        if (isClose) return [arms, st] as const;
        if (arms.length > 0) {
          // Allow optional comma between arms
          const hasComma = yield* check(st, "Delimiter", ",");
```

```typescript
const parsePattern = (s: ParseState): P<Ast.Pattern> =>
  Effect.gen(function* () {
    const t = yield* peek(s);
    // Wildcard
    if (tokenTag(t) === "Ident" && tokenValue(t) === "_") {
      const [tok, s1] = yield* advance(s);
      return [new Ast.WildcardPattern({ span: tokenSpan(tok) }), s1] as const;
    }
    // Constructor pattern (uppercase)
    if (tokenTag(t) === "TypeIdent") {
      const [tok, s1] = yield* advance(s);
      const tag = tokenValue(tok);
      // Collect sub-patterns (idents before -> or , or })
      const subPatterns: Ast.Pattern[] = [];
      const collectSubs = (st: ParseState): P<Ast.Pattern[]> =>
        Effect.gen(function* () {
          const next = yield* peek(st);
          // Stop at -> , } |
          if (tokenTag(next) === "Operator" && tokenValue(next) === "->")
            return [subPatterns, st] as const;
          if (
            tokenTag(next) === "Delimiter" &&
            (tokenValue(next) === "," || tokenValue(next) === "}" || tokenValue(next) === "|")
          )
            return [subPatterns, st] as const;
          if (tokenTag(next) === "EOF") return [subPatterns, st] as const;
          const [sub, st2] = yield* parsePattern(st);
          subPatterns.push(sub);
          return yield* collectSubs(st2);
        });
      const [patterns, s2] = yield* collectSubs(s1);
      return [
        new Ast.ConstructorPattern({
          tag,
          patterns,
          span: Span.merge(
            tokenSpan(tok),
            patterns.length > 0 ? patterns[patterns.length - 1].span : tokenSpan(tok),
          ),
        }),
        s2,
      ] as const;
    }
    // Literal patterns
    if (tokenTag(t) === "IntLit") {
      const [tok, s1] = yield* advance(s);
      const lit = new Ast.IntLiteral({ value: Number(tokenValue(tok)), span: tokenSpan(tok) });
      return [new Ast.LiteralPattern({ value: lit, span: tokenSpan(tok) }), s1] as const;
    }
    if (tokenTag(t) === "StringLit") {
      const [tok, s1] = yield* advance(s);
      const lit = new Ast.StringLiteral({ value: tokenValue(tok), span: tokenSpan(tok) });
      return [new Ast.LiteralPattern({ value: lit, span: tokenSpan(tok) }), s1] as const;
    }
    if (tokenTag(t) === "BoolLit") {
      const [tok, s1] = yield* advance(s);
      const lit = new Ast.BoolLiteral({ value: tokenBoolValue(tok), span: tokenSpan(tok) });
      return [new Ast.LiteralPattern({ value: lit, span: tokenSpan(tok) }), s1] as const;
    }
    // Binding pattern (lowercase ident, not _)
    if (tokenTag(t) === "Ident") {
      const [tok, s1] = yield* advance(s);
      return [new Ast.BindingPattern({ name: tokenValue(tok), span: tokenSpan(tok) }), s1] as const;
    }
    return yield* fail(`Expected pattern, got ${tokenDescription(t)}`, s);
  });
```

```typescript
const matchPattern = (pattern: Ast.Pattern, value: Value, env: Env): Option.Option<Env> =>
  Match.value(pattern).pipe(
    Match.tag("WildcardPattern", () => Option.some(env)),
    Match.tag("BindingPattern", (p) => Option.some(HashMap.set(env, p.name, value))),
    Match.tag("ConstructorPattern", (p) => {
      if (value._tag !== "Tagged" || value.tag !== p.tag) return Option.none();
      if (p.patterns.length !== value.fields.length) return Option.none();
      let currentEnv = env;
      for (let i = 0; i < p.patterns.length; i++) {
        const result = matchPattern(p.patterns[i], value.fields[i], currentEnv);
        if (Option.isNone(result)) return Option.none();
        currentEnv = result.value;
      }
      return Option.some(currentEnv);
    }),
    Match.tag("LiteralPattern", (p) => {
      // Evaluate the literal to a value and compare
      // Since LiteralPattern.value is an Expr (literal), we can match directly
      if (p.value._tag === "IntLiteral" && value._tag === "Num" && value.value === p.value.value)
        return Option.some(env);
      if (p.value._tag === "FloatLiteral" && value._tag === "Num" && value.value === p.value.value)
        return Option.some(env);
      if (p.value._tag === "StringLiteral" && value._tag === "Str" && value.value === p.value.value)
        return Option.some(env);
      if (p.value._tag === "BoolLiteral" && value._tag === "Bool" && value.value === p.value.value)
        return Option.some(env);
      if (p.value._tag === "UnitLiteral" && value._tag === "Unit") return Option.some(env);
      return Option.none();
    }),
    Match.exhaustive,
  );
```

```typescript
const parseImport = (s: ParseState): P<Ast.Import> =>
  Effect.gen(function* () {
    const [startTok, s1] = yield* expect(s, "Keyword", "from");
    // Parse module path: TypeIdent.TypeIdent.TypeIdent
    const [firstPart, s2] = yield* expect(s1, "TypeIdent");
    const path = [tokenValue(firstPart)];
    const collectPath = (st: ParseState): P<string[]> =>
      Effect.gen(function* () {
        const isDot = yield* check(st, "Operator", ".");
        if (!isDot) return [path, st] as const;
        const [, st2] = yield* advance(st);
        const [part, st3] = yield* expect(st2, "TypeIdent");
        path.push(tokenValue(part));
        return yield* collectPath(st3);
      });
    const [modulePath, s3] = yield* collectPath(s2);
    const [, s4] = yield* expect(s3, "Keyword", "import");
    const [, s5] = yield* expect(s4, "Delimiter", "{");
    // Parse comma-separated idents
    const names: string[] = [];
    const collectNames = (st: ParseState): P<string[]> =>
      Effect.gen(function* () {
        const isClose = yield* check(st, "Delimiter", "}");
        if (isClose) return [names, st] as const;
        if (names.length > 0) {
          const [, st2] = yield* expect(st, "Delimiter", ",");
          st = st2;
        }
        const [tok, st3] = yield* expect(st, "Ident");
        names.push(tokenValue(tok));
        return yield* collectNames(st3);
      });
    const [importNames, s6] = yield* collectNames(s5);
    const [endTok, s7] = yield* expect(s6, "Delimiter", "}");
    return [
      new Ast.Import({
        modulePath,
        names: importNames,
        span: Span.merge(tokenSpan(startTok), tokenSpan(endTok)),
      }),
      s7,
    ] as const;
  });

const parseExport = (s: ParseState): P<Ast.Export> =>
  Effect.gen(function* () {
    const [startTok, s1] = yield* expect(s, "Keyword", "export");
    const [, s2] = yield* expect(s1, "Delimiter", "{");
    const names: string[] = [];
    const collectNames = (st: ParseState): P<string[]> =>
      Effect.gen(function* () {
        const isClose = yield* check(st, "Delimiter", "}");
        if (isClose) return [names, st] as const;
        if (names.length > 0) {
          const [, st2] = yield* expect(st, "Delimiter", ",");
          st = st2;
        }
        const [tok, st3] = yield* expect(st, "Ident");
        names.push(tokenValue(tok));
        return yield* collectNames(st3);
      });
    const [exportNames, s3] = yield* collectNames(s2);
    const [endTok, s4] = yield* expect(s3, "Delimiter", "}");
    return [
      new Ast.Export({
        names: exportNames,
        span: Span.merge(tokenSpan(startTok), tokenSpan(endTok)),
```
